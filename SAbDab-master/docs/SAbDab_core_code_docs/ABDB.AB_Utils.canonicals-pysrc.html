168 -def _score_canonical(sequence, canonical): 

169      """ 
170      Function to score a canonical consensus sequence against a real sequence. 
171      Scored by top middle bottom conserved residues. 
172   
173      Capitals   => 90% conserved    - weight by 0.9   
174      Lower case => 20-90% conserved - weight by 0.55  
175      -          => <20% conserved   - weight by 0.05  
176   
177      """ 
178      consensus = canonical[2] 
179      top, middle, bottom = 0, 0, 0 
180      for i in range(len(consensus)): 
181          if consensus[i].isupper(): 
182              if consensus[i] != sequence[i].upper(): 
183                  return 0 
184              else: 
185                  top += 1 
186          elif consensus[i] == sequence[i].lower(): 
187              middle +=1  
188          elif consensus[i] == "-": # Anything might match. But want to give it some score otherwise at disadvantage over other canonicals with same lengths 
189              bottom += 1 
190      return 0.9*top + 0.55*middle, 0.05*bottom
