 18      """ 
 19      Search PATH for executable files with the given name. 
 20      
 21      On newer versions of MS-Windows, the PATHEXT environment variable will be 
 22      set to the list of file extensions for files considered executable. This 
 23      will normally include things like ".EXE". This fuction will also find files 
 24      with the given name ending with any of these extensions. 
 25   
 26      On MS-Windows the only flag that has any meaning is os.F_OK. Any other 
 27      flags will be ignored. 
 28      
 29      @type name: C{str} 
 30      @param name: The name for which to search. 
 31      
 32      @type flags: C{int} 
 33      @param flags: Arguments to L{os.access}. 
 34      
 35      @rtype: C{list} 
 36      @param: A list of the unique full paths to files found, in the 
 37      order in which they were found. 
 38      """ 
 39      result = [] 
 40      exts = filter(None, os.environ.get('PATHEXT', '').split(os.pathsep)) 
 41      path = os.environ.get('PATH', None) 
 42      if path is None: 
 43          return [] 
 44      for p in os.environ.get('PATH', '').split(os.pathsep): 
 45          p = os.path.join(p, name) 
 46          if os.access(p, flags): 
 47              result.append(p) 
 48          for e in exts: 
 49              pext = p + e 
 50              if os.access(pext, flags): 
 51                  result.append(pext) 
 52      return uniq(result)
